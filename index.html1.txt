    <script>
        const box = document.getElementById('giftBox');
        const overlay = document.getElementById('cardOverlay');
        const hint = document.getElementById('hintText');

        // Helper function to dynamically generate absolute bursting heart positions
        function createHeartParticle() {
            const heart = document.createElement('div');
            heart.classList.add('heart-particle');
            heart.innerHTML = '❤️';
            
            const x = (Math.random() - 0.5) * window.innerWidth * 0.8;
            const y = (Math.random() - 0.7) * window.innerHeight * 0.8;
            const size = Math.random() * 1.5 + 0.5;
            const rotation = Math.random() * 360;

            heart.style.setProperty('--x', `${x}px`);
            heart.style.setProperty('--y', `${y}px`);
            heart.style.setProperty('--s', size);
            heart.style.setProperty('--r', `${rotation}deg`);
            
            heart.style.left = '50%';
            heart.style.top = '50%';
            
            document.body.appendChild(heart);
            
            setTimeout(() => {
                heart.remove();
            }, 3000);
        }

        // Orchestrate the 15-second infinite automated lifecycle sequence
        function runAnimationTimeline() {
            // Stage 1: Reset state instantly (0 seconds)
            overlay.classList.remove('show');
            box.classList.remove('open');
            hint.textContent = "Watching the gift arrive... 🎁";
            hint.style.opacity = '1';

            // Stage 2: Trigger box opening and heart explosion (3 seconds)
            setTimeout(() => {
                box.classList.add('open');
                hint.style.opacity = '0';
                for(let i=0; i<40; i++) {
                    createHeartParticle();
                }
            }, 3000);

            // Stage 3: Smoothly transition love letter layout into view (4 seconds)
            setTimeout(() => {
                overlay.classList.add('show');
            }, 4000);

            // Stage 4: Maintain state display until closing down cleanly (14 seconds)
            setTimeout(() => {
                overlay.classList.remove('show');
                hint.textContent = "Wrapping up your gift... ✨";
                hint.style.opacity = '1';
            }, 14000);
        }

        // Initial launch kick-off
        runAnimationTimeline();

        // Loop the entire 15,000ms timeline seamlessly
        setInterval(runAnimationTimeline, 15000);
    </script>
